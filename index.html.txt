<script>
const layout = [
"XXXNIEDERLASSUNGSXXX",
"XXXAXXXXXXXXXXXXXXXX",
"SWITCHOVERKLAUSELXX",
"XXXGXXXXXXXXXXXXXXX",
"ASTGXXXXXXXXXXXXXXXX",
"XXBETRIEBSSTAETTEXX",
"XXXXXXXXXXXXXXXXXXX",
"FREISTELLUNGSMETHOD",
"XXXXXXXXXXXXXXXXXXX",
"ANRECHNUNGSMETHODEX",
"XXXXXXXXXXXXXXXXXXX",
"MOTIVTESTXXXXXXXXXX",
"XXZWISCHENEINKUENFT",
"XXXXXXXXXXXXXXXXXXX",
"CADBURYSCHWEPPESXXX",
"XXEUGHXXXXXXXXXXXXX",
"XXVORABENTSCHEIDUNG"
];

let inputs = [];
let time = 0;
let timer = setInterval(() => {
  time++;
  document.getElementById("time").innerText = time;
}, 1000);

// Bestzeit speichern
let best = localStorage.getItem("bestTime");
if (best) document.getElementById("best").innerText = best + "s";

const grid = document.getElementById("grid");

layout.forEach(row => {
  row.split("").forEach(c => {
    if (c === "X") {
      let div = document.createElement("div");
      div.className = "cell black";
      grid.appendChild(div);
    } else {
      let input = document.createElement("input");
      input.className = "cell";
      input.maxLength = 1;
      input.dataset.solution = c;

      input.addEventListener("input", () => move(input));

      inputs.push(input);
      grid.appendChild(input);
    }
  });
});

function move(current) {
  let index = inputs.indexOf(current);
  if (inputs[index + 1]) inputs[index + 1].focus();
}

function check() {
  let correct = 0;

  inputs.forEach(i => {
    if (i.value.toUpperCase() === i.dataset.solution) {
      i.classList.add("correct");
      i.classList.remove("wrong");
      correct++;
    } else {
      i.classList.add("wrong");
      i.classList.remove("correct");
    }
  });

  let percent = Math.round((correct / inputs.length) * 100);

  document.getElementById("result").innerHTML =
    "📊 Ergebnis: " + percent + "% richtig";

  if (percent === 100) {
    clearInterval(timer);

    document.getElementById("result").innerHTML +=
      "<br>🎉 Perfekt gelöst in " + time + " Sekunden!";

    if (!best || time < best) {
      localStorage.setItem("bestTime", time);
      document.getElementById("best").innerText = time + "s";
    }
  }
}

function reveal() {
  inputs.forEach(i => i.value = i.dataset.solution);
}
</script>
